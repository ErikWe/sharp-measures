    [Theory]
    [ClassData(typeof(#Quantity#Dataset))]
    public void Plus_Method_ShouldBeUnchanged(#Quantity# quantity)
    {
        #Quantity# result = quantity.Plus();

        Assert.Equal(quantity, result);
    }

    [Theory]
    [ClassData(typeof(#Quantity#Dataset))]
    public void Plus_Operator_ShouldBeUnchanged(#Quantity# quantity)
    {
        #Quantity# result = +quantity;

        Assert.Equal(quantity, result);
    }

    [ClassData(typeof(#Quantity#Dataset))]
    public void Negate_Method_ShouldBeScaledByMinusOne(#Quantity# quantity)

    }

    [ClassData(typeof(GenericDataset<#Quantity#Dataset, UnhandledDataset>))]
    public void MultiplyUnhandled_Operator_MagnitudeShouldEqualMultiplication(#Quantity# quantity, Unhandled factor)

        Assert.Equal(quantity.Magnitude * factor.Magnitude, resultLHS.Magnitude, 2);
        Assert.Equal(factor.Magnitude * quantity.Magnitude, resultRHS.Magnitude, 2);
